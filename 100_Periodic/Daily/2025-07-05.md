なぜ自分だけが長続きするものが好きなのかの理由がわかった。自分は、技術とか知識を身に付けることが好きで生きてきたけど、それらは努力することで成果が積み上がっていくものだった。だけど、多くの人が楽しんでいる。消費行動や人間関係は努力しても積み上がっていかない。だから、自分にとっては努力することで成果が積み上がっていくのが当たり前だけど、多くの人にとってはそこまで積み上がらないのが当たり前。人間関係の構築は努力することで成功確率が上がっていくのではなく、むしろ見かけ上下がっていってしまう。どういうことかって言うと例えば2分の1の成功を3回連続で引けば目標が達成できると考えて、試行を続けていたら連続ではなく5連続だったことにある日気づく時が来る。つまりもともと12.5%を引けば達成できると思っていたものが3.125%を引かないと達成できないことに気づく。ここで自分は努力しても達成確率が100%に近づかないことに大きな違和感を覚える。だけど、多くの人は努力が積み上がっていかないのが当たり前。だから、ここで見かけ上の確率が下がっていてもそこまで気にしない。だからこそ、なんとなく試行回数を稼ぐことができてその結果、3.125%をいつかひけるようになる。だから、自分が社会に適応していくでは努力することで、目標の達成確率を100%に近づけよって言う考え方ではなく、達成確率はそこまで高くないからこそ、試行回数を稼げるような考え方や行動を身に付けることが重要

const p = dv.pages("#Fleeting");

let rows =await Promise.all(
	p.map(async x => {

// コンテンツを処理
    let content = await dv.io.load(x.file.path);
    content = content.replace(/^---[\s\S]*?---\n/, ''); // YAMLフロントマターを除去
    content = content.replace(/\n+/g, ' ').trim(); // 改行を空白に変換
    content = content.replace(/^---\s*\[\[\d{4}-\d{2}-\d{2}\]\]\s*/m, ''); // 日付のリンクを除去
	content = content.replace(/^\s*!\[\[.*?\]\]\s*/gm, ''); // ![[FleetingDB_card]] のような埋め込みリンク行を除去

	  // 画像リンクを除去
    let noImages = content.replace(/!\[.*?\]\(.*?\)/g, "");
    // フロントマターを除去
    let noFrontmatter = noImages.replace(/^---[\s\S]*?---/, "");
    // 本文全体（必要ならtrimやsliceで調整）
    let body = noFrontmatter.trim();


   // プレビューテキストを生成
    const preview = body.length > 80 ? body.substring(0, 80) + "..." : body;
	return [x.file.link, preview || "", x.file.cday]
	})
);

dv.table(["title", "preview", "createdAt", ], rows);
